arug_dun01,1,2,1	script	怪物控制1#aru	4_DOG01,{
OnEnable:
	set .@callwhere,rand(1,4);
	if (.@callwhere == 1) {
		monster "arug_dun01",150,340,"庫布林",1980,1,"怪物控制1#aru::OnMyMobDead";
	}
	else if (.@callwhere == 2) {
		monster "arug_dun01",232,293,"庫布林",1980,1,"怪物控制1#aru::OnMyMobDead";
	}
	else if (.@callwhere == 3) {
		monster "arug_dun01",156,167,"庫布林",1980,1,"怪物控制1#aru::OnMyMobDead";
	}
	else {
		monster "arug_dun01",243,56,"庫布林",1980,1,"怪物控制1#aru::OnMyMobDead";
	}
	end;

OnKill:
	killmonster "arug_dun01","怪物控制1#aru::OnMyMobDead";
	end;

OnMyMobDead:
	if (mobcount("arug_dun01","怪物控制1#aru::OnMyMobDead") == 0) {
		mapannounce "arug_dun01", "庫布林: 吱唉唉唉!",bc_map,"0x99CC00";
		mapannounce "arug_dun01", "摩爾斯通: 好像聽到了庫布林的悲鳴聲音!! 誰在那裡? 庫布林怎麼樣了? 餵!",bc_map,"0x99CC00";
		donpcevent "矮人#aru_gd::OnEnable";
	}
	end;
}
